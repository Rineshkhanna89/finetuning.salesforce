	// Method to select "Logeshwari" from the Parent Account dropdown list

		return new VerifyEditContactPage();
	}

	// Method to verify toast message after account creation
	public VerifyNewContact VerifyAccountCreatedToastMessage(String expectedAccountName) {
		// Locate the toast message element
		String toastMessage = getElementText(locateElement(Locators.XPATH,
				"//span[contains(@class, 'toastMessage slds-text-heading--small forceActionsText')]"));

		// Log the toast message for debugging
		System.out.println("Toast Message: " + toastMessage);

		// Verify if the expected account name is in the toast message
		if (toastMessage.contains(expectedAccountName)) {
			reportStep("Toast message verified successfully. Account '" + expectedAccountName + "' was created.", "pass");
		} else {
			reportStep("Toast message verification failed. Expected '" + expectedAccountName + "' but found '" + toastMessage + "'.", "fail");
		}

		return this;
	}

}
